  8 require('ember-views/views/view');
  9 var get = Ember.get, set = Ember.set, meta = Ember.meta;
 10 var forEach = Ember.ArrayUtils.forEach;
 11 
 12 var childViewsProperty = Ember.computed(function() {
 13   return get(this, '_childViews');
 14 }).property('_childViews').cacheable();
 15 
 16 /**
 17   @class
 18   @extends Ember.View
 19   
 20   A `ContainerView` is an `Ember.View` subclass that allows for manual or programatic
 21   management of a view's `childViews` array that will correctly update the `ContainerView`
 22   instance's rendered DOM representation.
 23 
 24   ## Setting Initial Child Views
 25   The initial array of child views can be set in one of two ways. You can provide
 26   a `childViews` property at creation time that contains instance of `Ember.View`:
 27 
 28 
 29         aContainer = Ember.ContainerView.create({
 30           childViews: [Ember.View.create(), Ember.View.create()]
 31         })
 32 
 33   You can also provide a list of property names whose values are instances of `Ember.View`:
 34 
 35         aContainer = Ember.ContainerView.create({
 36           childViews: ['aView', 'bView', 'cView'],
 37           aView: Ember.View.create(),
 38           bView: Ember.View.create()
 39           cView: Ember.View.create()
 40         })
 41 
 42   The two strategies can be combined:
 43 
 44         aContainer = Ember.ContainerView.create({
 45           childViews: ['aView', Ember.View.create()],
 46           aView: Ember.View.create()
 47         })
 48 
 49   Each child view's rendering will be inserted into the container's rendered HTML in the same
 50   order as its position in the `childViews` property.
 51 
 52   ## Adding and Removing Child Views
 53   The views in a container's `childViews` array should be added and removed by manipulating
 54   the `childViews` property directly.
 55 
 56   To remove a view pass that view into a `removeObject` call on the container's `childViews` property. 
 57 
 58   Given an empty `<body>` the following code
 59 
 60         aContainer = Ember.ContainerView.create({
 61           classNames: ['the-container'],
 62           childViews: ['aView', 'bView'],
 63           aView: Ember.View.create({
 64             template: Ember.Handlebars.compile("A")
 65           }),
 66           bView: Ember.View.create({
 67             template: Ember.Handlebars.compile("B")
 68           })
 69         })
 70 
 71         aContainer.appendTo('body')
 72 
 73   Results in the HTML
 74 
 75         <div class="ember-view the-container">
 76           <div class="ember-view">A</div>
 77           <div class="ember-view">B</div>
 78         </div>
 79 
 80   Removing a view
 81 
 82         aContainer.get('childViews') // [aContainer.aView, aContainer.bView]
 83         aContainer.get('childViews').removeObject(aContainer.get('bView'))
 84         aContainer.get('childViews') // [aContainer.aView]
 85 
 86   Will result in the following HTML
 87 
 88         <div class="ember-view the-container">
 89           <div class="ember-view">A</div>
 90         </div>
 91 
 92 
 93   Similarly, adding a child view is accomplished by adding `Ember.View` instances to the 
 94   container's `childViews` property.
 95 
 96   Given an empty `<body>` the following code
 97 
 98         aContainer = Ember.ContainerView.create({
 99           classNames: ['the-container'],
100           childViews: ['aView', 'bView'],
101           aView: Ember.View.create({
102             template: Ember.Handlebars.compile("A")
103           }),
104           bView: Ember.View.create({
105             template: Ember.Handlebars.compile("B")
106           })
107         })
108 
109         aContainer.appendTo('body')
110 
111   Results in the HTML
112 
113         <div class="ember-view the-container">
114           <div class="ember-view">A</div>
115           <div class="ember-view">B</div>
116         </div>
117 
118   Adding a view
119 
120         AnotherViewClass = Ember.View.extend({
121           template: Ember.Handlebars.compile("Another view")
122         })
123 
124         aContainer.get('childViews') // [aContainer.aView, aContainer.bView]
125         aContainer.get('childViews').pushObject(AnotherViewClass.create())
126         aContainer.get('childViews') // [aContainer.aView, <AnotherViewClass instance>]
127 
128   Will result in the following HTML
129 
130         <div class="ember-view the-container">
131           <div class="ember-view">A</div>
132           <div class="ember-view">Another view</div>
133         </div>
134 
135 
136   Direct manipulation of childViews presence or absence in the DOM via calls to 
137   `remove` or `removeFromParent` or calls to a container's `removeChild` may not behave
138   correctly.
139 
140   Calling `remove()` on a child view will remove the view's HTML, but it will remain as part of its
141   container's `childView`s property.
142 
143   Calling `removeChild()` on the container will remove the passed view instance from the container's
144   `childView`s but keep its HTML within the container's rendered view.
145 
146   Calling `removeFromParent()` behaves as expected but should be avoided in favor of direct
147   manipulation of a container's `childViews` property.
148 
149         aContainer = Ember.ContainerView.create({
150           classNames: ['the-container'],
151           childViews: ['aView', 'bView'],
152           aView: Ember.View.create({
153             template: Ember.Handlebars.compile("A")
154           }),
155           bView: Ember.View.create({
156             template: Ember.Handlebars.compile("B")
157           })
158         })
159 
160         aContainer.appendTo('body')
161 
162   Results in the HTML
163 
164         <div class="ember-view the-container">
165           <div class="ember-view">A</div>
166           <div class="ember-view">B</div>
167         </div>
168 
169   Calling `aContainer.get('aView').removeFromParent()` will result in the following HTML
170 
171         <div class="ember-view the-container">
172           <div class="ember-view">A</div>
173           <div class="ember-view">B</div>
174         </div>
175 
176   And the `Ember.View` instance stored in `aContainer.aView` will be removed from `aContainer`'s
177   `childViews` array.
178 
179 
180   ## Templates and Layout
181   A `template`, `templateName`, `defaultTempalte`, `layout`, `layoutName` or `defaultLayout`
182   property on a container view will not result in the template or layout being rendered. 
183   The HTML contents of a `Ember.ContainerView`'s DOM representation will only be the rendered HTML
184   of its child views.
185 */
186 
187 Ember.ContainerView = Ember.View.extend({
188 
189   init: function() {
190     var childViews = get(this, 'childViews');
191     Ember.defineProperty(this, 'childViews', childViewsProperty);
192 
193     this._super();
194 
195     var _childViews = get(this, '_childViews');
196 
197     forEach(childViews, function(viewName, idx) {
198       var view;
199 
200       if ('string' === typeof viewName) {
201         view = get(this, viewName);
202         view = this.createChildView(view);
203         set(this, viewName, view);
204       } else {
205         view = this.createChildView(viewName);
206       }
207 
208       _childViews[idx] = view;
209     }, this);
210 
211     // Make the _childViews array observable
212     Ember.A(_childViews);
213 
214     // Sets up an array observer on the child views array. This
215     // observer will detect when child views are added or removed
216     // and update the DOM to reflect the mutation.
217     get(this, 'childViews').addArrayObserver(this, {
218       willChange: 'childViewsWillChange',
219       didChange: 'childViewsDidChange'
220     });
221   },
222 
223   /**
224     Instructs each child view to render to the passed render buffer.
225 
226     @param {Ember.RenderBuffer} buffer the buffer to render to
227     @private
228   */
229   render: function(buffer) {
230     this.forEachChildView(function(view) {
231       view.renderToBuffer(buffer);
232     });
233   },
234 
235   /**
236     When the container view is destroyed, tear down the child views
237     array observer.
238 
239     @private
240   */
241   willDestroy: function() {
242     get(this, 'childViews').removeArrayObserver(this, {
243       willChange: 'childViewsWillChange',
244       didChange: 'childViewsDidChange'
245     });
246 
247     this._super();
248   },
249 
250   /**
251     When a child view is removed, destroy its element so that
252     it is removed from the DOM.
253 
254     The array observer that triggers this action is set up in the
255     `renderToBuffer` method.
256 
257     @private
258     @param {Ember.Array} views the child views array before mutation
259     @param {Number} start the start position of the mutation
260     @param {Number} removed the number of child views removed
261   **/
262   childViewsWillChange: function(views, start, removed) {
263     if (removed === 0) { return; }
264 
265     var changedViews = views.slice(start, start+removed);
266     this.initializeViews(changedViews, null, null);
267 
268     this.invokeForState('childViewsWillChange', views, start, removed);
269   },
270 
271   /**
272     When a child view is added, make sure the DOM gets updated appropriately.
273 
274     If the view has already rendered an element, we tell the child view to
275     create an element and insert it into the DOM. If the enclosing container view
276     has already written to a buffer, but not yet converted that buffer into an
277     element, we insert the string representation of the child into the appropriate
278     place in the buffer.
279 
280     @private
281     @param {Ember.Array} views the array of child views afte the mutation has occurred
282     @param {Number} start the start position of the mutation
283     @param {Number} removed the number of child views removed
284     @param {Number} the number of child views added
285   */
286   childViewsDidChange: function(views, start, removed, added) {
287     var len = get(views, 'length');
288 
289     // No new child views were added; bail out.
290     if (added === 0) return;
291 
292     var changedViews = views.slice(start, start+added);
293     this.initializeViews(changedViews, this, get(this, 'templateData'));
294 
295     // Let the current state handle the changes
296     this.invokeForState('childViewsDidChange', views, start, added);
297   },
298 
299   initializeViews: function(views, parentView, templateData) {
300     forEach(views, function(view) {
301       set(view, '_parentView', parentView);
302       set(view, 'templateData', templateData);
303     });
304   },
305 
306   /**
307     Schedules a child view to be inserted into the DOM after bindings have
308     finished syncing for this run loop.
309 
310     @param {Ember.View} view the child view to insert
311     @param {Ember.View} prev the child view after which the specified view should
312                      be inserted
313     @private
314   */
315   _scheduleInsertion: function(view, prev) {
316     if (prev) {
317       prev.domManager.after(prev, view);
318     } else {
319       this.domManager.prepend(this, view);
320     }
321   }
322 });
323 
324 // Ember.ContainerView extends the default view states to provide different
325 // behavior for childViewsWillChange and childViewsDidChange.
326 Ember.ContainerView.states = {
327   parent: Ember.View.states,
328 
329   inBuffer: {
330     childViewsDidChange: function(parentView, views, start, added) {
331       var buffer = parentView.buffer,
332           startWith, prev, prevBuffer, view;
333 
334       // Determine where to begin inserting the child view(s) in the
335       // render buffer.
336       if (start === 0) {
337         // If views were inserted at the beginning, prepend the first
338         // view to the render buffer, then begin inserting any
339         // additional views at the beginning.
340         view = views[start];
341         startWith = start + 1;
342         view.renderToBuffer(buffer, 'prepend');
343       } else {
344         // Otherwise, just insert them at the same place as the child
345         // views mutation.
346         view = views[start - 1];
347         startWith = start;
348       }
349 
350       for (var i=startWith; i<start+added; i++) {
351         prev = view;
352         view = views[i];
353         prevBuffer = prev.buffer;
354         view.renderToBuffer(prevBuffer, 'insertAfter');
355       }
356     }
357   },
358 
359   hasElement: {
360     childViewsWillChange: function(view, views, start, removed) {
361       for (var i=start; i<start+removed; i++) {
362         views[i].remove();
363       }
364     },
365 
366     childViewsDidChange: function(view, views, start, added) {
367       // If the DOM element for this container view already exists,
368       // schedule each child view to insert its DOM representation after
369       // bindings have finished syncing.
370       var prev = start === 0 ? null : views[start-1];
371 
372       for (var i=start; i<start+added; i++) {
373         view = views[i];
374         this._scheduleInsertion(view, prev);
375         prev = view;
376       }
377     }
378   }
379 };
380 
381 Ember.ContainerView.states.inDOM = {
382   parentState: Ember.ContainerView.states.hasElement
383 };
384 
385 Ember.ContainerView.reopen({
386   states: Ember.ContainerView.states
387 });
