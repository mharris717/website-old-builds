  4 var indexOf = Ember.ArrayUtils.indexOf, indexesOf = Ember.ArrayUtils.indexesOf;

  6 Ember.Select = Ember.View.extend({
  7   tagName: 'select',
  8   template: Ember.Handlebars.compile(
  9     '{{#if prompt}}<option>{{prompt}}</option>{{/if}}' +
 10     '{{#each content}}{{view Ember.SelectOption contentBinding="this"}}{{/each}}'
 11   ),
 12   attributeBindings: ['multiple'],
 13 
 14   multiple: false,
 15   content: null,
 16   selection: null,
 17   prompt: null,
 18 
 19   optionLabelPath: 'content',
 20   optionValuePath: 'content',
 21 
 22   didInsertElement: function() {
 23     var selection = get(this, 'selection');
 24 
 25     if (selection) { this.selectionDidChange(); }
 26 
 27     this.change();
 28   },
 29 
 30   change: function() {
 31     if (get(this, 'multiple')) {
 32       this._changeMultiple();
 33     } else {
 34       this._changeSingle();
 35     }
 36   },
 37 
 38   selectionDidChange: Ember.observer(function() {
 39     var selection = get(this, 'selection'),
 40         isArray = Ember.isArray(selection);
 41     if (get(this, 'multiple')) {
 42       if (!isArray) {
 43         set(this, 'selection', Ember.A([selection]));
 44         return;
 45       }
 46       this._selectionDidChangeMultiple();
 47     } else {
 48       this._selectionDidChangeSingle();
 49     }
 50   }, 'selection'),
 51 
 52 
 53   _changeSingle: function() {
 54     var selectedIndex = this.$()[0].selectedIndex,
 55         content = get(this, 'content'),
 56         prompt = get(this, 'prompt');
 57 
 58     if (!content) { return; }
 59     if (prompt && selectedIndex === 0) { set(this, 'selection', null); return; }
 60 
 61     if (prompt) { selectedIndex -= 1; }
 62     set(this, 'selection', content.objectAt(selectedIndex));
 63   },
 64 
 65   _changeMultiple: function() {
 66     var options = this.$('option:selected'),
 67         prompt = get(this, 'prompt'),
 68         offset = prompt ? 1 : 0,
 69         content = get(this, 'content');
 70 
 71     if (!content){ return; }
 72     if (options) {
 73       var selectedIndexes = options.map(function(){
 74         return this.index - offset;
 75       }).toArray();
 76       set(this, 'selection', content.objectsAt(selectedIndexes));
 77     }
 78   },
 79 
 80   _selectionDidChangeSingle: function() {
 81     var el = this.$()[0],
 82         content = get(this, 'content'),
 83         selection = get(this, 'selection'),
 84         selectionIndex = indexOf(content, selection),
 85         prompt = get(this, 'prompt');
 86 
 87     if (prompt) { selectionIndex += 1; }
 88     if (el) { el.selectedIndex = selectionIndex; }
 89   },
 90 
 91   _selectionDidChangeMultiple: function() {
 92     var content = get(this, 'content'),
 93         selection = get(this, 'selection'),
 94         selectedIndexes = indexesOf(content, selection),
 95         prompt = get(this, 'prompt'),
 96         offset = prompt ? 1 : 0,
 97         options = this.$('option');
 98 
 99     if (options) {
100       options.each(function() {
101         this.selected = indexOf(selectedIndexes, this.index + offset) > -1;
102       });
103     }
104   }
105 
106 });

108 Ember.SelectOption = Ember.View.extend({
109   tagName: 'option',
110   template: Ember.Handlebars.compile("{{label}}"),
111   attributeBindings: ['value', 'selected'],
112 
113   init: function() {
114     this.labelPathDidChange();
115     this.valuePathDidChange();
116 
117     this._super();
118   },
119 
120   selected: Ember.computed(function() {
121     var content = get(this, 'content'),
122         selection = getPath(this, 'parentView.selection');
123     if (getPath(this, 'parentView.multiple')) {
124       return selection && indexOf(selection, content) > -1;
125     } else {
126       // Primitives get passed through bindings as objects... since
127       // `new Number(4) !== 4`, we use `==` below
128       return content == selection;
129     }
130   }).property('content', 'parentView.selection'),
131 
132   labelPathDidChange: Ember.observer(function() {
133     var labelPath = getPath(this, 'parentView.optionLabelPath');
134 
135     if (!labelPath) { return; }
136 
137     Ember.defineProperty(this, 'label', Ember.computed(function() {
138       return getPath(this, labelPath);
139     }).property(labelPath).cacheable());
140   }, 'parentView.optionLabelPath'),
141 
142   valuePathDidChange: Ember.observer(function() {
143     var valuePath = getPath(this, 'parentView.optionValuePath');
144 
145     if (!valuePath) { return; }
146 
147     Ember.defineProperty(this, 'value', Ember.computed(function() {
148       return getPath(this, valuePath);
149     }).property(valuePath).cacheable());
150   }, 'parentView.optionValuePath')
151 });
