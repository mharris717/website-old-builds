  7 require('ember-runtime/system/array_proxy');

  9 /**
 10   @class
 11 
 12   Ember.ArrayController provides a way for you to publish a collection of objects
 13   so that you can easily bind to the collection from a Handlebars #each helper,
 14   an Ember.CollectionView, or other controllers.
 15 
 16   The advantage of using an ArrayController is that you only have to set up
 17   your view bindings once; to change what's displayed, simply swap out the
 18   `content` property on the controller.
 19 
 20   For example, imagine you wanted to display a list of items fetched via an XHR
 21   request. Create an Ember.ArrayController and set its `content` property:
 22 
 23       MyApp.listController = Ember.ArrayController.create();
 24 
 25       $.get('people.json', function(data) {
 26         MyApp.listController.set('content', data);
 27       });
 28 
 29   Then, create a view that binds to your new controller:
 30 
 31       {{#each MyApp.listController}}
 32         {{firstName}} {{lastName}}
 33       {{/each}}
 34 
 35   Although you are binding to the controller, the behavior of this controller
 36   is to pass through any methods or properties to the underlying array. This
 37   capability comes from `Ember.ArrayProxy`, which this class inherits from.
 38 
 39   Note: As of this writing, `ArrayController` does not add any functionality
 40   to its superclass, `ArrayProxy`. The Ember team plans to add additional
 41   controller-specific functionality in the future, e.g. single or multiple
 42   selection support. If you are creating something that is conceptually a
 43   controller, use this class.
 44 
 45   @extends Ember.ArrayProxy
 46 */
 47 
 48 Ember.ArrayController = Ember.ArrayProxy.extend();
