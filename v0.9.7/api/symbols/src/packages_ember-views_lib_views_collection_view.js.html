  9 require('ember-runtime/system/string');

 11 var get = Ember.get, set = Ember.set, fmt = Ember.String.fmt;

 13 /**
 14   @class
 15   @since Ember 0.9
 16   @extends Ember.ContainerView
 17   
 18   `Ember.CollectionView` is an `Ember.View` descendent responsible for managing a
 19   collection (an array or array-like object) by maintaing a child view object and 
 20   associated DOM representation for each item in the array and ensuring that child
 21   views and their associated rendered HTML are updated when items in the array
 22   are added, removed, or replaced.
 23 
 24   ## Setting content
 25   The managed collection of objects is referenced as the `Ember.CollectionView` instance's
 26   `content` property.
 27 
 28       someItemsView = Ember.CollectionView.create({
 29         content: ['A', 'B','C']
 30       })
 31 
 32   The view for each item in the collection will have its `content` property set
 33   to the item.
 34 
 35   ## Specifying itemViewClass
 36   By default the view class for each item in the managed collection will be an instance
 37   of `Ember.View`. You can supply a different class by setting the `CollectionView`'s
 38   `itemViewClass` property.
 39 
 40   Given an empty `<body>` and the following code:
 41 
 42 
 43         someItemsView = Ember.CollectionView.create({
 44           classNames: ['a-collection'],
 45           content: ['A','B','C'],
 46           itemViewClass: Ember.View.extend({
 47             template: Ember.Handlebars.compile("the letter: {{content}}")
 48           })
 49         })
 50 
 51         someItemsView.appendTo('body')
 52 
 53   Will result in the following HTML structure
 54 
 55         <div class="ember-view a-collection">
 56           <div class="ember-view">the letter: A</div>
 57           <div class="ember-view">the letter: B</div>
 58           <div class="ember-view">the letter: C</div>
 59         </div>
 60 
 61 
 62   ## Automatic matching of parent/child tagNames
 63   Setting the `tagName` property of a `CollectionView` to any of 
 64   "ul", "ol", "table", "thead", "tbody", "tfoot", "tr", or "select" will result
 65   in the item views receiving an appropriately matched `tagName` property.
 66 
 67 
 68   Given an empty `<body>` and the following code:
 69 
 70         anUndorderedListView = Ember.CollectionView.create({
 71           tagName: 'ul',
 72           content: ['A','B','C'],
 73           itemViewClass: Ember.View.extend({
 74             template: Ember.Handlebars.compile("the letter: {{content}}")
 75           })
 76         })
 77 
 78         anUndorderedListView.appendTo('body')
 79 
 80   Will result in the following HTML structure
 81 
 82         <ul class="ember-view a-collection">
 83           <li class="ember-view">the letter: A</li>
 84           <li class="ember-view">the letter: B</li>
 85           <li class="ember-view">the letter: C</li>
 86         </ul>
 87 
 88   Additional tagName pairs can be provided by adding to `Ember.CollectionView.CONTAINER_MAP `
 89 
 90         Ember.CollectionView.CONTAINER_MAP['article'] = 'section'
 91 
 92 
 93   ## Empty View
 94   You can provide an `Ember.View` subclass to the `Ember.CollectionView` instance as its
 95   `emptyView` property. If the `content` property of a `CollectionView` is set to `null`
 96   or an empty array, an instance of this view will be the `CollectionView`s only child.
 97 
 98         aListWithNothing = Ember.CollectionView.create({
 99           classNames: ['nothing']
100           content: null,
101           emptyView: Ember.View.extend({
102             template: Ember.Handlebars.compile("The collection is empty")
103           })
104         })
105 
106         aListWithNothing.appendTo('body')
107 
108   Will result in the following HTML structure
109 
110         <div class="ember-view nothing">
111           <div class="ember-view">
112             The collection is empty
113           </div>
114         </div>
115 
116   ## Adding and Removing items
117   The `childViews` property of a `CollectionView` should not be directly manipulated. Instead,
118   add, remove, replace items from its `content` property. This will trigger
119   appropriate changes to its rendered HTML.
120 
121   ## Use in templates via the `{{collection}}` Ember.Handlebars helper
122   Ember.Handlebars provides a helper specifically for adding `CollectionView`s to templates.
123   See `Ember.Handlebars.collection` for more details
124   
125 */
126 Ember.CollectionView = Ember.ContainerView.extend(
127 /** @scope Ember.CollectionView.prototype */ {
128 
129   /**
130     A list of items to be displayed by the Ember.CollectionView.
131 
132     @type Ember.Array
133     @default null
134   */
135   content: null,
136 
137   /**
138     An optional view to display if content is set to an empty array.
139 
140     @type Ember.View
141     @default null
142   */
143   emptyView: null,
144 
145   /**
146     @type Ember.View
147     @default Ember.View
148   */
149   itemViewClass: Ember.View,
150 
151   /** @private */
152   init: function() {
153     var ret = this._super();
154     this._contentDidChange();
155     return ret;
156   },
157 
158   _contentWillChange: Ember.beforeObserver(function() {
159     var content = this.get('content');
160 
161     if (content) { content.removeArrayObserver(this); }
162     var len = content ? get(content, 'length') : 0;
163     this.arrayWillChange(content, 0, len);
164   }, 'content'),
165 
166   /**
167     @private
168 
169     Check to make sure that the content has changed, and if so,
170     update the children directly. This is always scheduled
171     asynchronously, to allow the element to be created before
172     bindings have synchronized and vice versa.
173   */
174   _contentDidChange: Ember.observer(function() {
175     var content = get(this, 'content');
176 
177     if (content) {
178       ember_assert(fmt("an Ember.CollectionView's content must implement Ember.Array. You passed %@", [content]), Ember.Array.detect(content));
179       content.addArrayObserver(this);
180     }
181 
182     var len = content ? get(content, 'length') : 0;
183     this.arrayDidChange(content, 0, null, len);
184   }, 'content'),
185 
186   willDestroy: function() {
187     var content = get(this, 'content');
188     if (content) { content.removeArrayObserver(this); }
189 
190     this._super();
191   },
192 
193   arrayWillChange: function(content, start, removedCount) {
194     // If the contents were empty before and this template collection has an
195     // empty view remove it now.
196     var emptyView = get(this, 'emptyView');
197     if (emptyView && emptyView instanceof Ember.View) {
198       emptyView.removeFromParent();
199     }
200 
201     // Loop through child views that correspond with the removed items.
202     // Note that we loop from the end of the array to the beginning because
203     // we are mutating it as we go.
204     var childViews = get(this, 'childViews'), childView, idx, len;
205 
206     len = get(childViews, 'length');
207 
208     var removingAll = removedCount === len;
209 
210     if (removingAll) {
211       this.invokeForState('empty');
212     }
213 
214     for (idx = start + removedCount - 1; idx >= start; idx--) {
215       childView = childViews[idx];
216       if (removingAll) { childView.removedFromDOM = true; }
217       childView.destroy();
218     }
219   },
220 
221   /**
222     Called when a mutation to the underlying content array occurs.
223 
224     This method will replay that mutation against the views that compose the
225     Ember.CollectionView, ensuring that the view reflects the model.
226 
227     This array observer is added in contentDidChange.
228 
229     @param {Array} addedObjects
230       the objects that were added to the content
231 
232     @param {Array} removedObjects
233       the objects that were removed from the content
234 
235     @param {Number} changeIndex
236       the index at which the changes occurred
237   */
238   arrayDidChange: function(content, start, removed, added) {
239     var itemViewClass = get(this, 'itemViewClass'),
240         childViews = get(this, 'childViews'),
241         addedViews = [], view, item, idx, len, itemTagName;
242 
243     if ('string' === typeof itemViewClass) {
244       itemViewClass = Ember.getPath(itemViewClass);
245     }
246 
247     ember_assert(fmt("itemViewClass must be a subclass of Ember.View, not %@", [itemViewClass]), Ember.View.detect(itemViewClass));
248 
249     len = content ? get(content, 'length') : 0;
250     if (len) {
251       for (idx = start; idx < start+added; idx++) {
252         item = content.objectAt(idx);
253 
254         view = this.createChildView(itemViewClass, {
255           content: item,
256           contentIndex: idx
257         });
258 
259         addedViews.push(view);
260       }
261     } else {
262       var emptyView = get(this, 'emptyView');
263       if (!emptyView) { return; }
264 
265       emptyView = this.createChildView(emptyView);
266       addedViews.push(emptyView);
267       set(this, 'emptyView', emptyView);
268     }
269 
270     childViews.replace(start, 0, addedViews);
271   },
272 
273   createChildView: function(view, attrs) {
274     view = this._super(view, attrs);
275 
276     var itemTagName = get(view, 'tagName');
277     var tagName = (itemTagName === null || itemTagName === undefined) ? Ember.CollectionView.CONTAINER_MAP[get(this, 'tagName')] : itemTagName;
278 
279     set(view, 'tagName', tagName);
280 
281     return view;
282   }
283 });
284 
285 /**
286   @static
287 
288   A map of parent tags to their default child tags. You can add
289   additional parent tags if you want collection views that use
290   a particular parent tag to default to a child tag.
291 
292   @type Hash
293   @constant
294 */
295 Ember.CollectionView.CONTAINER_MAP = {
296   ul: 'li',
297   ol: 'li',
298   table: 'tr',
299   thead: 'tr',
300   tbody: 'tr',
301   tfoot: 'tr',
302   tr: 'td',
303   select: 'option'
304 };
