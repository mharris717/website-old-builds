  7 var get = Ember.get, set = Ember.set;

  9 /**
 10   @class
 11 
 12   ## Overview
 13   
 14   This mixin provides properties and property observing functionality, core
 15   features of the Ember object model.
 16   
 17   Properties and observers allow one object to observe changes to a
 18   property on another object. This is one of the fundamental ways that
 19   models, controllers and views communicate with each other in an Ember
 20   application.
 21   
 22   Any object that has this mixin applied can be used in observer
 23   operations. That includes Ember.Object and most objects you will
 24   interact with as you write your Ember application.
 25 
 26   Note that you will not generally apply this mixin to classes yourself,
 27   but you will use the features provided by this module frequently, so it
 28   is important to understand how to use it.
 29   
 30   ## Using get() and set()
 31   
 32   Because of Ember's support for bindings and observers, you will always
 33   access properties using the get method, and set properties using the
 34   set method. This allows the observing objects to be notified and
 35   computed properties to be handled properly.
 36   
 37   More documentation about `get` and `set` are below.
 38   
 39   ## Observing Property Changes
 40 
 41   You typically observe property changes simply by adding the `observes`
 42   call to the end of your method declarations in classes that you write.
 43   For example:
 44 
 45       Ember.Object.create({
 46         valueObserver: function() {
 47           // Executes whenever the "value" property changes
 48         }.observes('value')
 49       });
 50     
 51   Although this is the most common way to add an observer, this capability
 52   is actually built into the Ember.Object class on top of two methods
 53   defined in this mixin: `addObserver` and `removeObserver`. You can use
 54   these two methods to add and remove observers yourself if you need to
 55   do so at runtime.
 56 
 57   To add an observer for a property, call:
 58 
 59       object.addObserver('propertyKey', targetObject, targetAction)
 60 
 61   This will call the `targetAction` method on the `targetObject` to be called
 62   whenever the value of the `propertyKey` changes.
 63   
 64   @extends Ember.Mixin
 65 */
 66 Ember.Observable = Ember.Mixin.create(/** @scope Ember.Observable.prototype */ {
 67 
 68   /** @private - compatibility */
 69   isObserverable: true,
 70 
 71   /**
 72     Retrieves the value of a property from the object.
 73 
 74     This method is usually similar to using object[keyName] or object.keyName,
 75     however it supports both computed properties and the unknownProperty
 76     handler.
 77     
 78     Because `get` unifies the syntax for accessing all these kinds
 79     of properties, it can make many refactorings easier, such as replacing a
 80     simple property with a computed property, or vice versa.
 81 
 82     ### Computed Properties
 83 
 84     Computed properties are methods defined with the `property` modifier
 85     declared at the end, such as:
 86 
 87           fullName: function() {
 88             return this.getEach('firstName', 'lastName').compact().join(' ');
 89           }.property('firstName', 'lastName')
 90 
 91     When you call `get` on a computed property, the function will be
 92     called and the return value will be returned instead of the function
 93     itself.
 94 
 95     ### Unknown Properties
 96 
 97     Likewise, if you try to call `get` on a property whose value is
 98     undefined, the unknownProperty() method will be called on the object.
 99     If this method returns any value other than undefined, it will be returned
100     instead. This allows you to implement "virtual" properties that are
101     not defined upfront.
102 
103     @param {String} key The property to retrieve
104     @returns {Object} The property value or undefined.
105   */
106   get: function(keyName) {
107     return get(this, keyName);
108   },
109 
110   /**
111     To get multiple properties at once, call getProperties
112     with a list of strings or an array:
113 
114           record.getProperties('firstName', 'lastName', 'zipCode'); // => { firstName: 'John', lastName: 'Doe', zipCode: '10011' }
115 
116    is equivalent to:
117 
118           record.getProperties(['firstName', 'lastName', 'zipCode']); // => { firstName: 'John', lastName: 'Doe', zipCode: '10011' }
119 
120     @param {String...|Array} list of keys to get
121     @returns {Hash}
122   */
123   getProperties: function() {
124     var ret = {};
125     var propertyNames = arguments;
126     if (arguments.length === 1 && Ember.typeOf(arguments[0]) === 'array') {
127       propertyNames = arguments[0];
128     }
129     for(var i = 0; i < propertyNames.length; i++) {
130       ret[propertyNames[i]] = get(this, propertyNames[i]);
131     }
132     return ret;
133   },
134 
135   /**
136     Sets the key equal to value.
137 
138     This method is generally very similar to calling object[key] = value or
139     object.key = value, except that it provides support for computed
140     properties, the unknownProperty() method and property observers.
141 
142     ### Computed Properties
143 
144     If you try to set a value on a key that has a computed property handler
145     defined (see the get() method for an example), then set() will call
146     that method, passing both the value and key instead of simply changing
147     the value itself. This is useful for those times when you need to
148     implement a property that is composed of one or more member
149     properties.
150 
151     ### Unknown Properties
152 
153     If you try to set a value on a key that is undefined in the target
154     object, then the unknownProperty() handler will be called instead. This
155     gives you an opportunity to implement complex "virtual" properties that
156     are not predefined on the object. If unknownProperty() returns
157     undefined, then set() will simply set the value on the object.
158 
159     ### Property Observers
160 
161     In addition to changing the property, set() will also register a
162     property change with the object. Unless you have placed this call
163     inside of a beginPropertyChanges() and endPropertyChanges(), any "local"
164     observers (i.e. observer methods declared on the same object), will be
165     called immediately. Any "remote" observers (i.e. observer methods
166     declared on another object) will be placed in a queue and called at a
167     later time in a coalesced manner.
168 
169     ### Chaining
170 
171     In addition to property changes, set() returns the value of the object
172     itself so you can do chaining like this:
173 
174           record.set('firstName', 'Charles').set('lastName', 'Jolley');
175 
176     @param {String} key The property to set
177     @param {Object} value The value to set or null.
178     @returns {Ember.Observable}
179   */
180   set: function(keyName, value) {
181     set(this, keyName, value);
182     return this;
183   },
184 
185   /**
186     To set multiple properties at once, call setProperties
187     with a Hash:
188 
189           record.setProperties({ firstName: 'Charles', lastName: 'Jolley' });
190 
191     @param {Hash} hash the hash of keys and values to set
192     @returns {Ember.Observable}
193   */
194   setProperties: function(hash) {
195     return Ember.setProperties(this, hash);
196   },
197 
198   /**
199     Begins a grouping of property changes.
200 
201     You can use this method to group property changes so that notifications
202     will not be sent until the changes are finished. If you plan to make a
203     large number of changes to an object at one time, you should call this
204     method at the beginning of the changes to begin deferring change
205     notifications. When you are done making changes, call endPropertyChanges()
206     to deliver the deferred change notifications and end deferring.
207 
208     @returns {Ember.Observable}
209   */
210   beginPropertyChanges: function() {
211     Ember.beginPropertyChanges();
212     return this;
213   },
214 
215   /**
216     Ends a grouping of property changes.
217 
218     You can use this method to group property changes so that notifications
219     will not be sent until the changes are finished. If you plan to make a
220     large number of changes to an object at one time, you should call
221     beginPropertyChanges() at the beginning of the changes to defer change
222     notifications. When you are done making changes, call this method to
223     deliver the deferred change notifications and end deferring.
224 
225     @returns {Ember.Observable}
226   */
227   endPropertyChanges: function() {
228     Ember.endPropertyChanges();
229     return this;
230   },
231 
232   /**
233     Notify the observer system that a property is about to change.
234 
235     Sometimes you need to change a value directly or indirectly without
236     actually calling get() or set() on it. In this case, you can use this
237     method and propertyDidChange() instead. Calling these two methods
238     together will notify all observers that the property has potentially
239     changed value.
240 
241     Note that you must always call propertyWillChange and propertyDidChange as
242     a pair. If you do not, it may get the property change groups out of order
243     and cause notifications to be delivered more often than you would like.
244 
245     @param {String} key The property key that is about to change.
246     @returns {Ember.Observable}
247   */
248   propertyWillChange: function(keyName){
249     Ember.propertyWillChange(this, keyName);
250     return this;
251   },
252 
253   /**
254     Notify the observer system that a property has just changed.
255 
256     Sometimes you need to change a value directly or indirectly without
257     actually calling get() or set() on it. In this case, you can use this
258     method and propertyWillChange() instead. Calling these two methods
259     together will notify all observers that the property has potentially
260     changed value.
261 
262     Note that you must always call propertyWillChange and propertyDidChange as
263     a pair. If you do not, it may get the property change groups out of order
264     and cause notifications to be delivered more often than you would like.
265 
266     @param {String} keyName The property key that has just changed.
267     @returns {Ember.Observable}
268   */
269   propertyDidChange: function(keyName) {
270     Ember.propertyDidChange(this, keyName);
271     return this;
272   },
273   
274   /**
275     Convenience method to call `propertyWillChange` and `propertyDidChange` in
276     succession.
277   
278     @param {String} keyName The property key to be notified about.
279     @returns {Ember.Observable}
280   */
281   notifyPropertyChange: function(keyName) {
282     this.propertyWillChange(keyName);
283     this.propertyDidChange(keyName);
284     return this;
285   },
286 
287   /**
288     Adds an observer on a property.
289 
290     This is the core method used to register an observer for a property.
291 
292     Once you call this method, anytime the key's value is set, your observer
293     will be notified. Note that the observers are triggered anytime the
294     value is set, regardless of whether it has actually changed. Your
295     observer should be prepared to handle that.
296 
297     You can also pass an optional context parameter to this method. The
298     context will be passed to your observer method whenever it is triggered.
299     Note that if you add the same target/method pair on a key multiple times
300     with different context parameters, your observer will only be called once
301     with the last context you passed.
302 
303     ### Observer Methods
304 
305     Observer methods you pass should generally have the following signature if
306     you do not pass a "context" parameter:
307 
308           fooDidChange: function(sender, key, value, rev);
309 
310     The sender is the object that changed. The key is the property that
311     changes. The value property is currently reserved and unused. The rev
312     is the last property revision of the object when it changed, which you can
313     use to detect if the key value has really changed or not.
314 
315     If you pass a "context" parameter, the context will be passed before the
316     revision like so:
317 
318           fooDidChange: function(sender, key, value, context, rev);
319 
320     Usually you will not need the value, context or revision parameters at
321     the end. In this case, it is common to write observer methods that take
322     only a sender and key value as parameters or, if you aren't interested in
323     any of these values, to write an observer that has no parameters at all.
324 
325     @param {String} key The key to observer
326     @param {Object} target The target object to invoke
327     @param {String|Function} method The method to invoke.
328     @returns {Ember.Object} self
329   */
330   addObserver: function(key, target, method) {
331     Ember.addObserver(this, key, target, method);
332   },
333 
334   /**
335     Remove an observer you have previously registered on this object. Pass
336     the same key, target, and method you passed to addObserver() and your
337     target will no longer receive notifications.
338 
339     @param {String} key The key to observer
340     @param {Object} target The target object to invoke
341     @param {String|Function} method The method to invoke.
342     @returns {Ember.Observable} receiver
343   */
344   removeObserver: function(key, target, method) {
345     Ember.removeObserver(this, key, target, method);
346   },
347 
348   /**
349     Returns true if the object currently has observers registered for a
350     particular key. You can use this method to potentially defer performing
351     an expensive action until someone begins observing a particular property
352     on the object.
353 
354     @param {String} key Key to check
355     @returns {Boolean}
356   */
357   hasObserverFor: function(key) {
358     return Ember.hasListeners(this, key+':change');
359   },
360 
361   /**
362     This method will be called when a client attempts to get the value of a
363     property that has not been defined in one of the typical ways. Override
364     this method to create "virtual" properties.
365     
366     @param {String} key The name of the unknown property that was requested.
367     @returns {Object} The property value or undefined. Default is undefined.
368   */
369   unknownProperty: function(key) {
370     return undefined;
371   },
372 
373   /**
374     This method will be called when a client attempts to set the value of a
375     property that has not been defined in one of the typical ways. Override
376     this method to create "virtual" properties.
377     
378     @param {String} key The name of the unknown property to be set.
379     @param {Object} value The value the unknown property is to be set to.
380   */
381   setUnknownProperty: function(key, value) {
382     this[key] = value;
383   },
384 
385   /**
386     This is like `get`, but allows you to pass in a dot-separated property
387     path.
388     
389         person.getPath('address.zip'); // return the zip
390         person.getPath('children.firstObject.age'); // return the first kid's age
391 
392     This reads much better than chained `get` calls.
393 
394     @param {String} path The property path to retrieve
395     @returns {Object} The property value or undefined.
396   */
397   getPath: function(path) {
398     return Ember.getPath(this, path);
399   },
400 
401   /**
402     This is like `set`, but allows you to specify the property you want to
403     set as a dot-separated property path.
404     
405         person.setPath('address.zip', 10011); // set the zip to 10011
406         person.setPath('children.firstObject.age', 6); // set the first kid's age to 6
407 
408     This is not as commonly used as `getPath`, but it can be useful.
409 
410     @param {String} path The path to the property that will be set
411     @param {Object} value The value to set or null.
412     @returns {Ember.Observable}
413   */
414   setPath: function(path, value) {
415     Ember.setPath(this, path, value);
416     return this;
417   },
418 
419   /**
420     Retrieves the value of a property, or a default value in the case that the property
421     returns undefined.
422     
423         person.getWithDefault('lastName', 'Doe');
424     
425     @param {String} keyName The name of the property to retrieve
426     @param {Object} defaultValue The value to return if the property value is undefined
427     @returns {Object} The property value or the defaultValue.
428   */
429   getWithDefault: function(keyName, defaultValue) {
430     return Ember.getWithDefault(this, keyName, defaultValue);
431   },
432 
433   /**
434     Set the value of a property to the current value plus some amount.
435     
436         person.incrementProperty('age');
437         team.incrementProperty('score', 2);
438     
439     @param {String} keyName The name of the property to increment
440     @param {Object} increment The amount to increment by. Defaults to 1
441     @returns {Object} The new property value
442   */
443   incrementProperty: function(keyName, increment) {
444     if (!increment) { increment = 1; }
445     set(this, keyName, (get(this, keyName) || 0)+increment);
446     return get(this, keyName);
447   },
448   
449   /**
450     Set the value of a property to the current value minus some amount.
451     
452         player.decrementProperty('lives');
453         orc.decrementProperty('health', 5);
454     
455     @param {String} keyName The name of the property to decrement
456     @param {Object} increment The amount to decrement by. Defaults to 1
457     @returns {Object} The new property value
458   */
459   decrementProperty: function(keyName, increment) {
460     if (!increment) { increment = 1; }
461     set(this, keyName, (get(this, keyName) || 0)-increment);
462     return get(this, keyName);
463   },
464 
465   /**
466     Set the value of a boolean property to the opposite of it's
467     current value.
468     
469         starship.toggleProperty('warpDriveEnaged');
470     
471     @param {String} keyName The name of the property to toggle
472     @returns {Object} The new property value
473   */
474   toggleProperty: function(keyName) {
475     set(this, keyName, !get(this, keyName));
476     return get(this, keyName);
477   },
478 
479   /**
480     Returns the cached value of a computed property, if it exists.
481     This allows you to inspect the value of a computed property
482     without accidentally invoking it if it is intended to be
483     generated lazily.
484 
485     @param {String} keyName
486     @returns {Object} The cached value of the computed property, if any
487   */
488   cacheFor: function(keyName) {
489     return Ember.cacheFor(this, keyName);
490   },
491 
492   /** @private - intended for debugging purposes */
493   observersForKey: function(keyName) {
494     return Ember.observersFor(this, keyName);
495   }
496 });
